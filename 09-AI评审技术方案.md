## 五、状态管理设计

### 5.1 Zustand Store (`stores/aiReviewStore.ts`)

```typescript
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import { AIReviewResult, AIReviewRequest, AIConfig, ReviewStatus } from '@/types/ai';
import { aiService } from '@/services/ai';

interface AIReviewStore {
  // 配置状态
  config: AIConfig;
  setConfig: (config: AIConfig) => Promise<void>;

  // 评审状态
  reviewStatus: ReviewStatus;
  currentResult: AIReviewResult | null;
  error: string | null;

  // 操作
  performReview: (request: AIReviewRequest) => Promise<void>;
  reReview: (request: AIReviewRequest) => Promise<void>;
  clearResult: () => void;
  ignoreIssue: (index: number) => void;
  restoreIssue: (index: number) => void;

  // 历史（按 worktreeId:branch 索引）
  getCachedReview: (worktreeId: string, branch: string) => AIReviewResult | undefined;
}

// 生成缓存键
const getCacheKey = (worktreeId: string, branch: string) =>
  `${worktreeId}:${branch}`;

export const aiReviewStore = create<AIReviewStore>()(
  persist(
    (set, get) => ({
      // 初始状态
      config: {
        provider: 'openai',
        apiKey: '',
        model: 'gpt-4o-mini',
        language: 'zh',
        autoReview: false,
      },
      reviewStatus: 'idle',
      currentResult: null,
      error: null,

      // 配置操作
      setConfig: async (config) => {
        await aiService.saveConfig(config);
        set({ config });
      },

      // 评审操作
      performReview: async (request) => {
        const { force, worktreePath, targetBranch } = request;

        // 检查缓存
        if (!force) {
          const cached = get().getCachedReview(worktreePath, targetBranch);
          if (cached) {
            set({ currentResult: cached, reviewStatus: 'success', error: null });
            return;
          }
        }

        set({ reviewStatus: 'loading', error: null });

        try {
          const response = await aiService.review(request);

          if (response.success && response.result) {
            set({
              currentResult: response.result,
              reviewStatus: 'success',
              error: null
            });
          } else {
            set({
              reviewStatus: 'error',
              error: response.error || '评审失败'
            });
          }
        } catch (err) {
          set({
            reviewStatus: 'error',
            error: err instanceof Error ? err.message : '未知错误'
          });
        }
      },

      reReview: (request) => {
        return get().performReview({ ...request, force: true });
      },

      clearResult: () => {
        set({ currentResult: null, reviewStatus: 'idle', error: null });
      },

      ignoreIssue: (index) => {
        const { currentResult } = get();
        if (!currentResult) return;

        const newIssues = [...currentResult.issues];
        if (newIssues[index]) {
          newIssues[index] = { ...newIssues[index], ignored: true };
          set({
            currentResult: { ...currentResult, issues: newIssues }
          });
        }
      },

      restoreIssue: (index) => {
        const { currentResult } = get();
        if (!currentResult) return;

        const newIssues = [...currentResult.issues];
        if (newIssues[index]) {
          newIssues[index] = { ...newIssues[index], ignored: false };
          set({
            currentResult: { ...currentResult, issues: newIssues }
          });
        }
      },

      // 获取缓存
      getCachedReview: (worktreeId, branch) => {
        // 从持久化存储中查找
        const storage = localStorage.getItem('ai-review-storage');
        if (!storage) return undefined;

        const data = JSON.parse(storage);
        const key = getCacheKey(worktreeId, branch);
        return data.state?.reviewHistory?.[key];
      },
    }),
    {
      name: 'ai-review-storage',
      partialize: (state) => ({
        config: state.config
      }),
    }
  )
);
```

### 6.1 AIReviewPanel 组件

```typescript
// components/DiffSidebar/AIReviewPanel.tsx

import { useState } from 'react';
import {
  Bot, AlertCircle, Lightbulb, CheckCircle,
  ChevronDown, ChevronUp, ExternalLink, RefreshCw, X
} from 'lucide-react';
import { AIReviewResult, ReviewIssue } from '@/types/ai';
import { aiReviewStore } from '@/stores/aiReviewStore';

interface AIReviewPanelProps {
  result: AIReviewResult | null;
  isLoading: boolean;
  error: string | null;
  onReReview: () => void;
  onNavigateToLine: (file: string, line: number) => void;
}

export function AIReviewPanel({
  result,
  isLoading,
  error,
  onReReview,
  onNavigateToLine
}: AIReviewPanelProps) {
  const [expandedSections, setExpandedSections] = useState({
    issues: true,
    improvements: true,
    highlights: true,
  });

  const toggleSection = (section: keyof typeof expandedSections) => {
    setExpandedSections(prev => ({ ...prev, [section]: !prev[section] }));
  };

  // 加载中状态
  if (isLoading) {
    return (
      <div className="flex items-center justify-center py-8">
        <div className="flex flex-col items-center gap-2">
          <Bot className="w-8 h-8 text-blue-500 animate-pulse" />
          <span className="text-sm text-gray-500">AI 正在分析代码...</span>
        </div>
      </div>
    );
  }

  // 错误状态
  if (error) {
    return (
      <div className="p-4 bg-red-50 dark:bg-red-900/20 rounded-lg">
        <div className="flex items-center gap-2 text-red-700 dark:text-red-400">
          <AlertCircle className="w-5 h-5" />
          <span className="font-medium">评审失败</span>
        </div>
        <p className="mt-1 text-sm text-red-600 dark:text-red-300">{error}</p>
        <button
          onClick={onReReview}
          className="mt-3 px-3 py-1.5 text-sm bg-red-100 dark:bg-red-800 text-red-700 dark:text-red-300 rounded hover:bg-red-200 dark:hover:bg-red-700"
        >
          重试
        </button>
      </div>
    );
  }

  // 无结果状态
  if (!result) {
    return null;
  }

  const activeIssues = result.issues.filter(i => !i.ignored);

  return (
    <div className="space-y-4">
      {/* 头部 */}
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <Bot className="w-5 h-5 text-blue-500" />
          <span className="font-medium text-gray-900 dark:text-white">
            AI 评审结果
          </span>
        </div>
        <button
          onClick={onReReview}
          className="flex items-center gap-1 text-sm text-gray-500 hover:text-blue-600"
        >
          <RefreshCw className="w-4 h-4" />
          重新评审
        </button>
      </div>

      {/* 潜在问题 */}
      {activeIssues.length > 0 && (
        <ReviewSection
          title="潜在问题"
          count={activeIssues.length}
          icon={<AlertCircle className="w-4 h-4 text-orange-500" />}
          isExpanded={expandedSections.issues}
          onToggle={() => toggleSection('issues')}
        >
          <IssuesList
            issues={activeIssues}
            onNavigate={onNavigateToLine}
          />
        </ReviewSection>
      )}

      {/* 改进建议 */}
      {result.improvements.length > 0 && (
        <ReviewSection
          title="改进建议"
          count={result.improvements.length}
          icon={<Lightbulb className="w-4 h-4 text-blue-500" />}
          isExpanded={expandedSections.improvements}
          onToggle={() => toggleSection('improvements')}
        >
          <ImprovementsList improvements={result.improvements} />
        </ReviewSection>
      )}

      {/* 亮点 */}
      {result.highlights.length > 0 && (
        <ReviewSection
          title="亮点"
          count={result.highlights.length}
          icon={<CheckCircle className="w-4 h-4 text-green-500" />}
          isExpanded={expandedSections.highlights}
          onToggle={() => toggleSection('highlights')}
        >
          <HighlightsList highlights={result.highlights} />
        </ReviewSection>
      )}
    </div>
  );
}
```

use reqwest::{Client, header::{HeaderMap, HeaderValue, AUTHORIZATION, CONTENT_TYPE}};
use serde_json::{json, Value};

const MAX_DIFF_TOKENS: usize = 8000;
const OPENAI_API_URL: &str = "https://api.openai.com/v1/chat/completions";
const CLAUDE_API_URL: &str = "https://api.anthropic.com/v1/messages";
const OLLAMA_API_URL: &str = "http://localhost:11434/api/generate";

/// AI 服务
pub struct AIService {

impl AIService {
    pub fn new() -> Self {

    /// 测试 AI 连接
    pub async fn test_connection(&self, config: &AIConfig) -> anyhow::Result<bool> {
        let prompt = "Hello, this is a test. Please respond with 'OK' only.";

        match config.provider {
            AIProvider::OpenAI => {
                let response = self.call_openai(config, prompt).await?;
                Ok(response.to_lowercase().contains("ok"))
            }
            AIProvider::Claude => {
                let response = self.call_claude(config, prompt).await?;
                Ok(response.to_lowercase().contains("ok"))
            }
            AIProvider::Ollama => {
                let response = self.call_ollama(config, prompt).await?;
                Ok(!response.is_empty())
            }
            AIProvider::Custom => {
                let response = self.call_custom(config, prompt).await?;
                Ok(!response.is_empty())
            }
        }
    }

    /// 执行代码评审
    pub async fn review_code(&self, config: &AIConfig, diff: &str, stats: &DiffStats) -> anyhow::Result<AIReviewResult> {
        // 截断过长的 diff
        let truncated_diff = self.truncate_diff(diff);

        // 构建 prompt
        let prompt = self.build_review_prompt(&truncated_diff, stats, &config.language);

        // 调用 AI API
        let response = match config.provider {
            AIProvider::OpenAI => self.call_openai(config, &prompt).await?,
            AIProvider::Claude => self.call_claude(config, &prompt).await?,
            AIProvider::Ollama => self.call_ollama(config, &prompt).await?,
            AIProvider::Custom => self.call_custom(config, &prompt).await?,
        };

        // 解析响应
        let result = self.parse_review_response(&response)?;

        Ok(result)
    }

    /// 调用 OpenAI API
    async fn call_openai(&self, config: &AIConfig, prompt: &str) -> anyhow::Result<String> {
        let url = OPENAI_API_URL;

        let mut headers = HeaderMap::new();
        headers.insert(AUTHORIZATION, HeaderValue::from_str(&format!("Bearer {}", config.api_key))?);
        headers.insert(CONTENT_TYPE, HeaderValue::from_static("application/json"));

        let body = json!({
            "model": config.model,
            "messages": [{"role": "user", "content": prompt}],
            "temperature": 0.3,
            "max_tokens": 4000
        });

        let response = self.client
            .post(url)
            .headers(headers)
            .json(&body)
            .send()
            .await?;

        if !response.status().is_success() {
            let error_text = response.text().await?;
            return Err(anyhow::anyhow!("OpenAI API error: {}", error_text));
        }

        let json: Value = response.json().await?;
        let content = json["choices"][0]["message"]["content"]
            .as_str()
            .unwrap_or_default()
            .to_string();

        Ok(content)
    }

    /// 调用 Claude API
    async fn call_claude(&self, config: &AIConfig, prompt: &str) -> anyhow::Result<String> {
        let url = CLAUDE_API_URL;

        let mut headers = HeaderMap::new();
        headers.insert("x-api-key", HeaderValue::from_str(&config.api_key)?);
        headers.insert("anthropic-version", HeaderValue::from_static("2023-06-01"));
        headers.insert(CONTENT_TYPE, HeaderValue::from_static("application/json"));

            .unwrap_or_default()

    /// 截断过长的 diff
    fn truncate_diff(&self, diff: &str) -> String {
        // 简单估算：1 token ≈ 4 字符
        let max_chars = MAX_DIFF_TOKENS * 4;

        if diff.len() <= max_chars {

    fn parse_review_response(&self, response: &str) -> anyhow::Result<AIReviewResult> {
        // 尝试提取 JSON
        let json_str = self.extract_json(response);

        let parsed: Value = serde_json::from_str(&json_str)?;

        let issues: Vec<ReviewIssue> = parsed["issues"]
            .as_array()
            .unwrap_or(&vec![])
            .iter()
            .filter_map(|v| serde_json::from_value(v.clone()).ok())
            .collect();

        let improvements: Vec<ReviewImprovement> = parsed["improvements"]
            .as_array()
            .unwrap_or(&vec![])
            .iter()
            .filter_map(|v| serde_json::from_value(v.clone()).ok())
            .collect();

        let highlights: Vec<ReviewHighlight> = parsed["highlights"]
            .as_array()
            .unwrap_or(&vec![])
            .iter()
            .filter_map(|v| serde_json::from_value(v.clone()).ok())
            .collect();

        Ok(AIReviewResult {

            target_branch: String::new(), // 由调用方填充

            improvements,
            highlights,

    /// 从响应中提取 JSON
    fn extract_json(&self, response: &str) -> String {
        // 尝试找到 JSON 块
        if let Some(start) = response.find('{') {
            if let Some(end) = response.rfind('}') {
                return response[start..=end].to_string();
            }
        }
        response.to_string()
    }

```rust
        result: Some(result),
        error: None,
    })
```
